drop table if exists emergenza, missione, luogo, coinvolgimento, luogo_specifico , svolgimento , 
report,evento_saliente, missione_di_campo , sezione, membro , domicilio , recapito , qualifica ,
specializzazione , dipendente , volontario,capo_missione , team , coordinamento , 
partecipazione_corrente , partecipazione_passata , ferimento , decesso, realizzazione_corrente, 
realizzazione_passata , missione_esplorativa , autorizzazione;
drop domain if exists positive_integer cascade;

create domain positive_integer as integer check(value  > 0);

create table emergenza(
	codice_emergenza positive_integer Primary Key,
	nome varchar(30) NOT NULL UNIQUE,
	descrizione varchar(200) NOT NULL,
	data_di_inizio date,
	data_di_fine date,
	gravità numeric(1,0) NOT NULL,
	check (data_di_fine > data_di_inizio),
	check(gravità > 0 )
);

create table missione (
	codice_missione positive_integer Primary Key,
	data_di_inizio date NOT NULL,
	data_fine_presunta date NOT NULL,
	descrizione varchar(200) NOT NULL,
	num_membri_coinvolti integer NOT NULL default 0,
	emergenza positive_integer references emergenza on delete restrict on update cascade NOT NULL,
	check (data_fine_presunta > data_di_inizio),
	check(num_membri_coinvolti >= 0 )
);

create table luogo (
	nome varchar(20) Primary Key
);

create table coinvolgimento (
	emergenza positive_integer references emergenza on delete restrict on update cascade,
	luogo varchar(20) references luogo on delete restrict on update restrict,
	primary key (emergenza, luogo)
);

create table luogo_specifico(
	nome varchar(20),
	luogo_di_appartenenza varchar(20) references luogo on delete restrict on update restrict,
	primary key (nome, luogo_di_appartenenza)
);

create table svolgimento(
	luogo_specifico varchar(20),
	luogo_di_appartenenza varchar(20),
	missione positive_integer references missione on delete restrict on update cascade,
	foreign key (luogo_specifico, luogo_di_appartenenza) 
	references luogo_specifico(nome,luogo_di_appartenenza) on delete restrict on update restrict,
	primary key (luogo_specifico, luogo_di_appartenenza, missione)
);

create table report(
	missione positive_integer  primary key references missione on delete restrict on update cascade,
	costo_totale numeric(8,2) NOT NULL,
	successo_o_insuccesso numeric(1,0) NOT NULL,
	motivazione_variazione_data varchar(100),
	data_fine_effettiva date NOT NULL,
	check (costo_totale > 0 ),
	check (successo_o_insuccesso >= 0)
);

create table evento_saliente(
	codice_evento serial primary key,
	nome varchar(100) NOT NULL,
	missione positive_integer references report on delete restrict on update cascade NOT NULL
);

create table missione_di_campo(
	missione positive_integer primary key references missione on delete restrict on update cascade,
	obiettivo varchar(30) NOT NULL,
	tipologia varchar(30) NOT NULL,
	valutazione_rischio numeric(1,0) NOT NULL,
	check (valutazione_rischio >= 0 )
);

create table sezione(
	sede varchar(30) primary key,
	numero_di_uffici positive_integer NOT NULL ,
	numero_membri integer NOT NULL default 0 ,
	check (numero_membri >= 0 )
);

create table membro (
	codice_membro positive_integer primary key,
	national_identity_number varchar(25) NOT NULL UNIQUE,
	professione varchar(30) NOT NULL,
	nome varchar(30) NOT NULL,
	cognome varchar(30) NOT NULL,
	data_di_nascita date NOT NULL,
	Via varchar(30),
	Civico varchar(5),
	CAP varchar(15),
	sezione varchar(30) references sezione on delete restrict on update cascade NOT NULL
);

create table domicilio (
	codice_domicilio positive_integer primary key,
	Via varchar(30) NOT NULL,
	Civico varchar(5) NOT NULL,
	CAP varchar(15) NOT NULL,
	UNIQUE (Via,Civico,CAP)
);

create table recapito(
	membro positive_integer references membro on delete restrict on update cascade,
	domicilio positive_integer references domicilio on delete cascade on update cascade,
	primary key (membro, domicilio)
);

create table qualifica (
	titolo varchar(30) primary key
);

create table specializzazione (
	membro positive_integer references membro on delete restrict on update cascade,
	qualifica varchar(30) references qualifica on delete cascade on update cascade,
	primary key (membro, qualifica)
	
);

create table dipendente(
	codice_membro positive_integer primary key references membro 
	on delete cascade on update cascade,
	stipendio numeric(7,2) NOT NULL default 0, 
	anni_di_servizio positive_integer default 0 ,
	check (stipendio>0)
);

create table volontario(
	codice_membro positive_integer primary key references membro 
	on delete restrict on update cascade,
	data_inizio_disponibilità date NOT NULL,
	data_fine_disponibilità date NOT NULL,
	check (data_fine_disponibilità > data_inizio_disponibilità)
);

create table capo_missione(
	dipendente positive_integer primary key references dipendente 
	on delete restrict on update cascade
);

create table team(
	codice_team positive_integer primary key,
	capo_missione positive_integer references capo_missione 
	on delete restrict on update cascade NOT NULL
);

create table coordinamento(
	team positive_integer references team  on delete restrict on update cascade,
	membro positive_integer references membro  on delete restrict on update cascade,
	tipo varchar(30) NOT NULL,
	primary key (team,membro)
);

create table partecipazione_corrente(
	data_inizio date,
	membro positive_integer references membro on delete restrict on update cascade,
	team positive_integer references team on delete restrict on update cascade,
	primary key (data_inizio,membro)
	
);

create table partecipazione_passata(
	data_inizio date,
	membro positive_integer references membro on delete restrict on update cascade,
	team positive_integer references team on delete restrict on update cascade,
	data_fine date NOT NULL,
	primary key (data_inizio,membro),
	check (data_fine > data_inizio)
);

create table ferimento(
	codice_ferimento serial primary key,
	data_evento date NOT NULL,
	motivazione varchar(30) NOT NULL,
	data_guarigione date,
	missione positive_integer references missione on delete restrict on update cascade,
	membro positive_integer references membro on delete restrict on update cascade,
	check (data_guarigione > data_evento),
	UNIQUE(missione,data_evento, membro)
);

create table decesso(
	membro positive_integer primary key references membro on delete restrict on update cascade,
	missione positive_integer references missione on delete restrict on update cascade,
	data_decesso date NOT NULL,
	motivazione varchar(30) NOT NULL
);

create table realizzazione_corrente(
	team positive_integer primary key references team on delete restrict on update cascade ,
	missione_di_campo positive_integer references missione_di_campo 
	on delete restrict on update cascade NOT NULL, 
	data_inizio_attività date NOT NULL
);

create table realizzazione_passata(
	team positive_integer primary key references team on delete restrict on update cascade,
	missione_di_campo positive_integer references missione_di_campo 
	on delete restrict on update cascade NOT NULL, 
	data_inizio_attività date NOT NULL,
	data_fine_attività date NOT NULL,
	check (data_fine_attività > data_inizio_attività)
	
);

create table missione_esplorativa(
	missione positive_integer primary key references missione on delete restrict on update cascade,
	esito boolean default NULL,
	team positive_integer references team on delete restrict on update cascade NOT NULL UNIQUE
);

create table autorizzazione(
	missione_esplorativa positive_integer references missione_esplorativa 
	on delete restrict on update cascade,
	missione_di_campo positive_integer references missione_di_campo 
	on delete restrict on update cascade,
	primary key (missione_esplorativa,missione_di_campo)
);
